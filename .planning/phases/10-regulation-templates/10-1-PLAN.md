```python
"""EU AI Act regulation mapping and risk classification."""

from dataclasses import dataclass, field
from typing import Any

@dataclass
class ArticleMapping:
    article: str
    title: str
    requirement: str
    report_section: str
    status: str  # "addressed" | "partially_addressed" | "not_addressed"
    evidence: str

@dataclass  
class RegulationReport:
    regulation: str  # "eu_ai_act"
    risk_classification: str  # "high" | "medium" | "low"
    article_mappings: list[ArticleMapping]
    disclaimer: str

class RegulationMapper:
    """Maps FairCheck analysis results to EU AI Act articles."""
    
    DISCLAIMER = (
        "DISCLAIMER: This report is generated by automated software and is provided "
        "as guidance only. It does not constitute legal certification of compliance "
        "with the EU AI Act or any other regulation. Organizations should seek "
        "qualified legal counsel to verify regulatory compliance."
    )
    
    def map(self, report_data) -> RegulationReport:
        """Map analysis results to EU AI Act articles."""
        risk = self._classify_risk(report_data.analysis_results)
        mappings = self._build_mappings(report_data)
        return RegulationReport(
            regulation="eu_ai_act",
            risk_classification=risk,
            article_mappings=mappings,
            disclaimer=self.DISCLAIMER,
        )
    
    def _classify_risk(self, analysis: dict) -> str:
        results = analysis.get("results", {})
        has_fail = has_warn = False
        for attr_data in results.values():
            for m_data in attr_data.get("metrics", {}).values():
                s = m_data.get("status", "")
                if s == "fail": has_fail = True
                elif s == "warning": has_warn = True
        if has_fail: return "high"
        if has_warn: return "medium"
        return "low"
    
    def _build_mappings(self, data) -> list[ArticleMapping]:
        has_oversight = data.oversight is not None
        has_model_card = bool(data.training_data_description)
        has_analysis = bool(data.analysis_results.get("results"))
        
        return [
            ArticleMapping("Article 9", "Risk Management System", ...),
            ArticleMapping("Article 10", "Data Governance", ...),
            ArticleMapping("Article 13", "Transparency", ...),
            ArticleMapping("Article 14", "Human Oversight", ...),
            ArticleMapping("Article 17", "Quality Management", ...),
        ]
```

Key behaviors:
- Risk classification: High if any fail, Medium if any warn, Low if all pass
- Each article produces an `ArticleMapping` with status and evidence text
- Status is "addressed" if relevant data exists, "not_addressed" if absent
- Article 14 links to oversight data from WEB-07
</action>

</acceptance_criteria>
</task>
